    public static int getLCM(int x, int y) {
        int max = x>y ? x : y;
        int r=1;
        for (int i=max; i <= x*y; i++) {
            System.out.println(i);
            if (i%x==0 && i%y==0) {
                r = i;
                break;
            }
        }
        return r;
